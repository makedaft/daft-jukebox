% DAC Left
\createDAC{dacLeft}{(20, -10.5)}

\bendconnect{1}{-60}{(mcupin14)}{(dacLeftpinCLK)}; \dotOnFirstBend;

% Audio jack
\createTRRSAudioJack{audioJack}{(29, -3)}

\bendconnect{-1}{20}{(dacLeftpinOut+)}{($(audioJackpinR1)+(-1.7, 0)$)} to[R=$R_{out}$] (audioJackpinR1);
