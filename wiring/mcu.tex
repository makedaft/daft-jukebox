\newcommand{\createMCU}[2] {
  % Parameters:
  % #1: ESP32 node name
  % #2: Position
  \def\name{#1}
  \def\pos{#2}

  \node [draw, fill=white, text=black, minimum width=4cm, minimum height=7cm] (\name) at \pos {
    \begin{tabular}{c}
      \texttt{\textbf{ESP32}} \\
      \texttt{(\name)}
    \end{tabular}
  };

  \def\pinLabelsLeft{{"GND1", "36", "39", "34", "35", "32", "33", "25", "26", "27", "14", "12", "13", "GND2", "Vin"}}
  \def\pinLabelsRight{{"23", "22", "1", "3", "21", "19", "18", "5", "17", "16", "4", "2", "15", "GND3", "3v3"}}
  \def\pinSetSize{6}

  \setupBoardPins{\name}{\pinSetSize}{\pinLabelsLeft}{\pinLabelsRight}
}
