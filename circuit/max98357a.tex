\newcommand{\createDAC}[2] {
  % Parameters:
  % #1: MAX98357A node name
  % #2: Position
  \def\name{#1}
  \def\pos{#2}

  \node [draw, fill=white, text=black, minimum width=5.5cm, minimum height=6cm] (\name) at \pos {
    \begin{tabular}{c}
      \texttt{\textbf{MAX98357A}} \\
      \texttt{(\name)}
    \end{tabular}
  };

  \def\pinLabelsLeft{{"LRC", "CLK", "Din", "Gain", "SD", "GND", "Vin"}}
  \def\pinLabelsRight{{"Out+", "Out-"}}
  \pgfmathsetmacro{\pinSetSize}{5}

  \setupBoardPins{\name}{\pinSetSize}{\pinLabelsLeft}{\pinLabelsRight}
}
